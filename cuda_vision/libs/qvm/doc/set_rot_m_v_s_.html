Boost QVM

set_rot(m,v,s)

#include <boost/qvm/m.hpp>

namespace qvm
{
    //Only enabled if:
    //  is_m<A>::value && m_traits<A>::rows>=3 &&
    //  m_traits<A>::rows==m_traits<A>::cols &&
    //  is_v<B>::value && v_traits<B>::dim==3
    template <class A>
    void set_rot( A & a, B const & axis, typename v_traits<B>::scalar_type angle );
}

Effects:

As if:

assign(a,rot_m<m_traits<A>::rows>(axis,angle));
See also: boost/qvm/m.hpp
