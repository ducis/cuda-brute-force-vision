Boost QVM

operator/(q,s)

#include <boost/qvm/q.hpp>

namespace boost
{
    namespace qvm
    {
        //Only enabled if: is_q<A>::value
        template <class A>
        typename deduce_q<A>::type
        operator/( A const & a, q_traits<A>::scalar_type s );
    }
}

Returns:

A quaternion that is the result of dividing the quaternion a by the scalar s.
